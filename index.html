export default { async fetch(request, env) { const url = new URL(request.url); const cnpj = url.searchParams.get("cnpj"); const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, chave-api-dados", }; if (request.method === "OPTIONS") return new Response(null, { headers: corsHeaders }); if (!cnpj) return new Response(JSON.stringify({ error: "CNPJ ausente" }), { status: 400, headers: corsHeaders }); // Gerar datas (Portal exige DD/MM/AAAA) const hoje = new Date(); const umAnoAtras = new Date(); umAnoAtras.setFullYear(hoje.getFullYear() - 1); const fmtData = (d) => d.toLocaleDateString('pt-BR'); const dataInicial = fmtData(umAnoAtras); const dataFinal = fmtData(hoje); // Montando a URL com o filtro de data obrigatório const apiUrl = `https://api.portaldatransparencia.gov.br/api-de-dados/despesas/documentos-por-favorecido?codigoBeneficiario=${cnpj}&dataInicial=${dataInicial}&dataFinal=${dataFinal}&pagina=1`; try { const response = await fetch(apiUrl, { method: "GET", headers: { "chave-api-dados": env.CHAVE_PORTAL, "Accept": "application/json" } }); const data = await response.json(); return new Response(JSON.stringify(data), { status: response.status, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } catch (err) { return new Response(JSON.stringify({ error: "Erro de conexão", details: err.message }), { status: 500, headers: corsHeaders }); } } };
